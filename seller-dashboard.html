// ================== LOGIN SYSTEM ================== let currentUser = null; function handleLogin() { const email = document.getElementById("email").value.trim(); const password = document.getElementById("password").value.trim(); if (!email || !password) { alert("Please enter email and password"); return; } // Demo accounts if (email === "seller@kenya.com" && password === "123456") { currentUser = { name: "Mike Seller", role: "seller" }; localStorage.setItem("currentUser", JSON.stringify(currentUser)); window.location.href = "seller-dashboard.html"; } else if (email === "customer@kenya.com" && password === "123456") { currentUser = { name: "John Doe", role: "customer" }; localStorage.setItem("currentUser", JSON.stringify(currentUser)); window.location.href = "index.html"; } else { alert("Wrong email or password!\n\nTry these:\n\nSeller: seller@kenya.com / 123456\nCustomer: customer@kenya.com / 123456"); } } // Check if already logged in function checkLoginStatus() { const savedUser = localStorage.getItem("currentUser"); if (savedUser) { currentUser = JSON.parse(savedUser); } } function logout() { localStorage.removeItem("currentUser"); window.location.href = "login.html"; } function showLogin() { window.location.href = "login.html"; } // Make functions available window.handleLogin = handleLogin; window.logout = logout; window.showLogin = showLogin; window.checkLoginStatus = checkLoginStatus; document.addEventListener("DOMContentLoaded", checkLoginStatus);
